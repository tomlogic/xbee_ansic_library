179 //
180 // Gateway capabilities.
181 // This is a set of features supported by the network and gateway.
182 // This set may depend on the hardware and firmware versions of
183 // the gateway and gateway radio currently in use.
184 // Individual nodes may not support all of these features.
185 //
186 
187 // Advanced addressing: ability to send and receive to any endpoint and
188 // cluster ID on remote nodes. Without this capability, only a single
189 // endpoint and cluster ID (e.g. remote serial port) may be used on each node.
190 #define ZB_CAP_ADV_ADDR 0x00000001
191 
192 // Ability to access ZigBee device objects
193 #define ZB_CAP_ZDO 0x00000002
194 
195 // Ability to access Digi device objects on remote nodes.
196 // DDO access to the gateway radio is always available.
197 #define ZB_CAP_REMOTE_DDO 0x00000004
198 
199 // Ability to update firmware on the gateway radio.
200 #define ZB_CAP_GW_FW 0x00000008
201 
202 // Ability to update firmware on remote nodes via the gateway.
203 #define ZB_CAP_REMOTE_FW 0x00000010
204 
205 // Supports ZigBee mesh networking
206 #define ZB_CAP_ZIGBEE 0x00000020
207 
208 // Supports ZigBee Pro/2007 mesh networking
209 #define ZB_CAP_ZBPRO 0x00000040
210 
211 // Supports DigiMesh networking
212 #define ZB_CAP_DIGIMESH 0x00000080
213 
214 // Supports parent/child relationship
215 #define ZB_CAP_CHILDREN 0x00000100
216 
217 // Supports ZigBee Smart Energy profile
218 #define ZB_CAP_SE 0x00000200
